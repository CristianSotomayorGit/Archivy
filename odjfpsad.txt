from bs4 import BeautifulSoup, Tag
import sys

def extract_sections(html_file):
    with open(html_file, 'r', encoding='utf-8') as file:
        soup = BeautifulSoup(file, 'html.parser')

    

    elements = soup.find_all()

    with open('newContent.html', 'w', encoding='utf-8') as file:

        for i, elem in enumerate(elements):

            char_num = 0


            if not isinstance(elem, tuple):
                if isinstance(elem, Tag) and not elem.find():  # Check if elem is a leaf Tag
                    if 'name' in elem.attrs:

                        elem_name = elem['name']

                        file.write('</div>')
                        file.write('<div id="' + elem_name + '">')

                    if len(elem.get_text()) != 0:
                        elem_text = elem.get_text()
                        file.write(elem_text + '\n')


    

def main():
    html_file2 = 'content.html'
    extract_sections(html_file2)

if __name__ == "__main__":
    main()
